\hypertarget{growth-from-metabolic-energy-inflow}{%
\subsubsection{Growth from metabolic energy
inflow}\label{growth-from-metabolic-energy-inflow}}

\begin{itemize}
\tightlist
\item
  \(B\) = incoming rate of energy flow, which is the average resting
  metabolic rate of the whole organism at time \(t\)
\item
  \(B_c\) = the metabolic rate of a single cell
\item
  \(E_c\) = the metabolic energy required to create a cell
\item
  \(N_c\) = the total number of cells
\item
  \(\sum\limits_c\) = over all types of tissue, assuming a typical cell
  as the fundamental unit
\item
  tells us that dedication to growth \textbf{AFTER} metabolism is a
  fundamental axiom of energy/mass conservation equation
\item growth is equal to surplus energy because assumed that energy is optimally allocated to growth after miscellaneous costs 
  \begin{align*}
    B &= \sum_c \Bigg[N_{c}B_{c} + E_{c}\frac{dN_{c}}{dt}\Bigg] \\
    B &= N_{c}B_{c} + E_{c}\frac{dN_{c}}{dt} \\
    \frac{dN_{c}}{dt} &= \frac{B - N_{c}B_{c}}{E_{c}} = \frac{\text{Surplus energy}}{\text{Cost of creating single cell}} \\
    &= \text{multiply by mass of single cell} \\
    \Bigg(\frac{dN_{c}}{dt}\Bigg)m_c &= \Bigg(\frac{B - N_{c}B_{c}}{E_{c}}\Bigg)m_c \\
    \frac{dm}{dt} &= \Bigg(\frac{Bm_c}{E_c}\Bigg) - \Bigg(\frac{N_{c}m_{c}B_{c}}{E_c}\Bigg) \\
    \frac{dm}{dt} &= \Bigg(\frac{Bm_c}{E_c}\Bigg) - \Bigg(\frac{m_{tot}B_{c}}{E_c}\Bigg) \\
    \frac{dm}{dt} &= \Bigg(\frac{m_c}{E_c}\Bigg)B - \Bigg(\frac{B_{c}}{E_c}\Bigg)m_{tot} \\
    \text{sub } B &= B_{0}m^{3/4}  \\
    \frac{dm}{dt} &= \frac{m_c}{E_c}B_{0}m^{3/4} - \frac{B_{c}}{E_c}m_{tot} \\
    \text{sub } a &= \frac{m_c}{E_c}B_{0}  \\
    b &= \frac{B_{c}}{E_c} \\
    \frac{dm}{dt} &= am^{3/4} - bm \\
  \end{align*}
\subsubsection{Surplus is reduced by efficiency of use (ingestion
  versus energy flow into branching network)}
\item
  \(I\) = consumption rate \(\big(\frac{mass}{time}\big)\)
\item
  \(t_I\) = ``characteristic'' time period for ``sufficient'' energy
  gain
\item
  \(t_I << t_{growth}\)
\item
  Need to tackle this time period, we know
  \(t \propto \kappa m^{\gamma}\) i.e.~bigger things can spend longer
  foraging, thus intake more (absolute not relative)
\item
  But their digestion is then slower
\item
  reasonable bounding of digestion times?
\item
  Timescale at level of digestion is different to timescale of growth
\item
  You may pass a threshold where you are digesting at maximum rate
  meaning that you are limited to \(3/4\) exponent because limited by
  fractal dimension of capillary network \begin{align*}
    I_{tot} &= \int_{t_{start}}^{t_{end}} I dt \\
    \frac{m_c}{E_c}B_{0}m^{3/4} &= am^{3/4} \\
    \frac{am^{3/4}}{m} &= am^{\frac{3}{4} - 1} \\
    \frac{am^{3/4}}{m} &= am^{-\frac{1}{4}} \\
    I_{tot}\epsilon\Big(am^{-\frac{1}{4}}\Big) \\
  \end{align*} 
\subsubsection{Can also introduce loss to reproduction}
  \begin{align*}
    \frac{dm}{dt} &= I_{tot}\epsilon\Big(am^{-\frac{1}{4}}\Big) - bm - cm^{\rho} \\
  \end{align*}
\end{itemize}
